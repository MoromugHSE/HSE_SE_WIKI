            /*проблема в том, чтобы вывести свое имя в несколько строк(ну из символов оно точно будет

            {
                " **** ",

                "*     ",
                "*    *",
                " **** "
            };
            string[] a =
            {
                "   *  ",
                "  **  ",
                " *  * ",
                " **** ",
                "*    *"
            };
            string[] sh =
            {
                "*  *  *",
                "*  *  *",
                "*  *  *",
                "*  *  *",
                "*******"
            };
            Console.WriteLine(); //перенос строки

                Console.WriteLine(s[i] + " " + a[i] + " " + sh[i] + " " + a[i]);

                " ****     *   *  *  *    *  ",

                "*       *  *  *  *  *  *  * ",
                "*    *  ****  *  *  *  **** ",
                " ****  *    * ******* *    *"
            };
            for (var i = 0; i < 5; i++)
            {
                Console.WriteLine(sasha[i]);
            }
            Console.WriteLine();
            //и наконец форматированный вывод, опять копируем строки
            Console.WriteLine($" {ch}{ch}{ch}{ch}     {ch}   {ch}  {ch}  {ch}    {ch}  ");

using System;

namespace Task04
{
    class Program
    {
        static double Vvod(string name)
        {
            /*на одном моменте я стригерился на повторяющийся код и сделал фунцию, которая запрашивает
             (для данной задачи) ввести ту или иную физическую величину(name - ее название) и возвращает 
             её уже вещественным числом 
             Для тех кто ещё не занет про фунции:
             static - функция статична(я пока хз, вообщем так пишите) _ double - тип возвращаемого значения
             потом мдет имя Vvod, а в скобках указывается тип и локальное имя перемнной(ну в нашем случае 
             это строка, ибо Console.WriteLine())
             после return пишем то, что возвращается функцией
             */
            Console.WriteLine($"Введите значение {name}: ");
            double rtn;
            double.TryParse(Console.ReadLine(), out rtn);
            return rtn;
        }
        
        static void Main(string[] args)
        {
            //Сделаем сначала с простым вводом в стобик, а потом в строку
            double U, R;
            Console.WriteLine("Введите значение U: ");
            double.TryParse(Console.ReadLine(), out U);
            Console.WriteLine("Введите значение R: ");
            double.TryParse(Console.ReadLine(), out R);
            //на этом моменте мне показалось что идет повторени и я стригерился на создание функции
            Console.WriteLine("Значение напряжения P: {0}", Math.Pow(U, 2)/R);
            
            
            //тоже самое с функцией
            U =Vvod("U");
            R =Vvod("R");
            Console.WriteLine("Значение напряжения P: {0}", Math.Pow(U, 2)/R);
            // намного проще и красивее выглядит, всего-то надо разобраться с функциями
            //теперь ввод в одну строку 
            Console.WriteLine("Введите значение U и R: ");
            string[] value = Console.ReadLine().Split();
            // .Split(*что-то*) разделяет строку на элементы, разделитель между элементами *что-то*
            // но по дефолту это пробел
            double.TryParse(value[0], out U);
            double.TryParse(value[1], out R);
            Console.WriteLine("Значение напряжения P: {0}", Math.Pow(U, 2)/R);
            //ну и для красоты можно было бы заменить на (Math.Pow(U, 2)/R).ToString("F2")
        }
    }
}
