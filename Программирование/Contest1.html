        {
            int number;
            string num = Console.ReadLine();
            const int k = 4;
            //создал отдельную переменную для строки для более простой проверки числа на то, что оно 4х знач

                //сделал преобразование через строки, чтобы легче получить значения цифр числа, но можно 
                //было их получить через

                {
                    Console.WriteLine("True");
                }
                else
                {
                    Console.WriteLine("False");
                }
            }
            else
            {
                Console.WriteLine("Incorrect input");
            }
        }
    }
}

        {

            if (double.TryParse(catet1, out ct1) && double.TryParse(catet2, out ct2) && ct1>0 && ct2>0)

        {
            Gipotenusa(Console.ReadLine(), Console.ReadLine());

using System;

namespace I
{
    class Program
    {
        static void Main(string[] args)
        {
            double num;
            if (double.TryParse(Console.ReadLine(), out num))
            {
                /*записал в отдельную переменную дробную часть, чтобы с ней непосредственно работать*/
                double b = Math.Abs(num) - (int) Math.Abs(num);
                if (b != 0.5)
                {
                    Console.WriteLine(Math.Round(num));//просто округляем, если рне равноудалено
                }
                else if (b == 0.5 && Math.Abs((num + 0.5))%2==1)//проверяем какое из ближайших чисел нечетное
                {
                    Console.WriteLine(num + 0.5);
                }
                else if (b == 0.5 && Math.Abs((num - 0.5))%2==1)//это условие можно было не писать
                {
                    Console.WriteLine(num - 0.5);
                }
            }
            else
            {
                Console.WriteLine("Incorrect input");
            }
        }
    }
}
